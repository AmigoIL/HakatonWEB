import telegram
from telegram.ext import Updater, CommandHandler
import schedule
import time
import threading

# Ваш токен от BotFather
TOKEN = '8103581992:AAGqM_MRH9tqzlsO5p_3WIDtz4Xh4rEOE5Q'
CHANNEL_ID = '@penkagoroskop'

# Сообщение для отправки
MESSAGE = 'Привет, это запланированное сообщение!'

# Функция для отправки сообщения в канал
def send_scheduled_message(context):
    context.bot.send_message(chat_id=CHANNEL_ID, text=MESSAGE)

# Функция для запуска расписания
def run_schedule():
    while True:
        schedule.run_pending()
        time.sleep(1)

def start(update, context):
    update.message.reply_text("Бот запущен и готов к отправке запланированных сообщений.")

def main():
    # Настройка бота и обработчика команд
    updater = Updater(TOKEN, use_context=True)
    dp = updater.dispatcher
    dp.add_handler(CommandHandler("start", start))

    # Расписание для отправки сообщений, например, каждый день в 10:00
    schedule.every().day.at("10:00").do(lambda: send_scheduled_message(dp.context))

    # Запускаем поток с расписанием
    threading.Thread(target=run_schedule).start()

    # Запуск бота
    updater.start_polling()
    updater.idle()

if __name__ == '__main__':
    main()
